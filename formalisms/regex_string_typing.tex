\begin{mathpar}
\inferrule[Constant Example]
{
\String \in \Sigma^*
}
{
\Delta \vdash \String : \String
}

\inferrule[Userdef Example]
{
\Delta' \vdash \String : \Regex
}
{
\Delta' \cup \{(\Regex,\UserDefined)\} \vdash \String : \UserDefined
}

\inferrule[Concat Example]
{
\Delta \vdash \String_1 : \Regex_1\\
\Delta \vdash \String_2 : \Regex_2\\
}
{
\Delta \vdash \String_1\String_2 : \Regex_1 \Regex_2
}

\inferrule[Or Example]
{
\Delta \vdash \String : \Regex_1\\
}
{
\Delta \vdash \String : \Regex_1 \GBar \Regex_2\\
}

\inferrule[Empty Star]
{
}
{
\Delta \vdash \epsilon : \Regex^*
}

\inferrule[Nonempty Star]
{
\Delta \vdash \String_1 : \Regex^*\\
\Delta \vdash \String_2 : \Regex\\
}
{
\Delta \vdash \String_1\String_2 : \Regex^*
}
\end{mathpar}
