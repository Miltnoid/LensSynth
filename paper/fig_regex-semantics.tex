\item $\IterateLens{\Lens}.putr(\epsilon) = \epsilon$

\item $\OrLens{\Lens_1}{\Lens_2}.putr(\String) = \Lens_1(\String)$ if $\String \in dom(\Lens_1)$ or $\Lens_2(\String)$ if $\String \in dom(\Lens_2)$
